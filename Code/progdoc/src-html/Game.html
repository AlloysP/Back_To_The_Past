001/**
002 *  Classe principale du jeux "Back to Past". 
003 * "Back to Past" est un jeux textuel avec interface graphique.  L'utilisateur peut voyager dans le temps
004 * en changeant d'images à chaque nouvelle "pièce".
005 * 
006 *  Pour jouer au jeux, crée une instance de cette classe.
007 * 
008 *  Cette classe implémente/instancie les objets nécessaires au démarrage du jeux et démarre et stop le jeux.
009 * 
010 * @author  Petit Alloys
011 * @version 3.0
012 */
013
014public class Game
015{
016        private UserInterface aGui;
017        private GameEngine aEngine;
018
019    /**
020     * Crée le jeux et initialise la map des "salles".
021     */
022    public Game() 
023    {
024        this.aEngine = new GameEngine();
025                this.aGui = new UserInterface(aEngine);
026                this.aEngine.setGUI(aGui);
027    }
028}
